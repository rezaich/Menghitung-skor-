# Menghitung-skor-
fun main() {     
  var skor = -1      
  
  if (skor &lt;= 100 && skor >= 81) {
    println("A")     
    } else if (skor &lt;= 80 &amp;&amp; skor >= 71) {         
        println("B")     
    } else if (skor &lt;= 70 &amp;&amp; skor >= 61) {         
        println("C")     
    } else if (skor >= 0 &amp;&amp; skor &lt;= 60) {         
        println("D")     
    } else {         
        print("Skor tidak diketahui")     
    } 
}
